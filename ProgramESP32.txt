#include <Arduino.h>
#include <ESP32Servo.h>
#include <WiFi.h>
#include <HTTPClient.h>
#include <ArduinoJson.h>


const char* ssid       = "NazwaTwojejSieci";      
const char* password   = "HasloDoWiFi";             
const char* serverBase = "http://wikkur16.pythonanywhere.com";
const int   DEVICE_ID  = 1;

const unsigned long SEND_INTERVAL_MS = 10000; // co 10 sekund
unsigned long lastSendTime = 0;

const int SERVO_PIN = 18;

const int MOTOR_PWM = 25;
const int MOTOR_IN1 = 26;
const int MOTOR_IN2 = 27;

const int LDR_TOP_LEFT_PIN     = 32;
const int LDR_TOP_RIGHT_PIN    = 33;
const int LDR_BOTTOM_LEFT_PIN  = 34;
const int LDR_BOTTOM_RIGHT_PIN = 35;

Servo solarServo;
int currentServoAngle = 90;


int hysteresisHighServo = 150;
int hysteresisLowServo  = 80;
int hysteresisHighMotor = 150;
int hysteresisLowMotor  = 80;

bool servoMovingVert  = false;
bool motorMovingHoriz = false;

int lastTL = 0, lastTR = 0, lastBL = 0, lastBR = 0;

const int LDR_SAMPLES = 8;

int readAdcAveraged(int pin, int samples) {
  long sum = 0;
  for (int i = 0; i < samples; i++) {
    sum += analogRead(pin);
  }
  return (int)(sum / samples);
}


void moveServoToAngle(int angle) {
  angle = constrain(angle, 0, 180);
  currentServoAngle = angle;
  solarServo.write(angle);
}

#define MOTOR_STOP  0
#define MOTOR_LEFT  1
#define MOTOR_RIGHT 2

void setMotor(uint8_t dir, uint8_t pwmValue) {
  pwmValue = constrain(pwmValue, 0, 255);
  switch (dir) {
    case MOTOR_STOP:
      analogWrite(MOTOR_PWM, 0);
      digitalWrite(MOTOR_IN1, LOW);
      digitalWrite(MOTOR_IN2, LOW);
      break;
    case MOTOR_LEFT:
      digitalWrite(MOTOR_IN1, HIGH);
      digitalWrite(MOTOR_IN2, LOW);
      analogWrite(MOTOR_PWM, pwmValue);
      break;
    case MOTOR_RIGHT:
      digitalWrite(MOTOR_IN1, LOW);
      digitalWrite(MOTOR_IN2, HIGH);
      analogWrite(MOTOR_PWM, pwmValue);
      break;
  }
}


void updateTracking() {
  int tl = readAdcAveraged(LDR_TOP_LEFT_PIN,     LDR_SAMPLES);
  int tr = readAdcAveraged(LDR_TOP_RIGHT_PIN,    LDR_SAMPLES);
  int bl = readAdcAveraged(LDR_BOTTOM_LEFT_PIN,  LDR_SAMPLES);
  int br = readAdcAveraged(LDR_BOTTOM_RIGHT_PIN, LDR_SAMPLES);

  lastTL = tl; lastTR = tr; lastBL = bl; lastBR = br;

  int leftSum   = tl + bl;
  int rightSum  = tr + br;
  int topSum    = tl + tr;
  int bottomSum = bl + br;


  int diffVert = topSum - bottomSum;
  int absDiffV = abs(diffVert);

  if (!servoMovingVert) {
    if (absDiffV > hysteresisHighServo) servoMovingVert = true;
  } else {
    if (absDiffV < hysteresisLowServo)  servoMovingVert = false;
  }

  if (servoMovingVert) {
    if (diffVert > 0)      moveServoToAngle(currentServoAngle + 1);
    else if (diffVert < 0) moveServoToAngle(currentServoAngle - 1);
  }

  int diffHoriz = leftSum - rightSum;
  int absDiffH  = abs(diffHoriz);

  if (!motorMovingHoriz) {
    if (absDiffH > hysteresisHighMotor) motorMovingHoriz = true;
  } else {
    if (absDiffH < hysteresisLowMotor)  motorMovingHoriz = false;
  }

  if (motorMovingHoriz) {
    uint8_t pwm = 180;
    if (diffHoriz > 0)      setMotor(MOTOR_LEFT,  pwm);
    else if (diffHoriz < 0) setMotor(MOTOR_RIGHT, pwm);
  } else {
    setMotor(MOTOR_STOP, 0);
  }
}


void sendToServer() {
  if (WiFi.status() != WL_CONNECTED) {
    Serial.println("Brak WiFi, pomijam wysylke");
    return;
  }

  HTTPClient http;
  String url = String(serverBase) + "/device/" + DEVICE_ID + "/mes1";
  http.begin(url);
  http.addHeader("Content-Type", "application/json");

  String json = "{";
  json += "\"foto1\":"  + String(lastTL) + ",";
  json += "\"foto2\":"  + String(lastTR) + ",";
  json += "\"foto3\":"  + String(lastBL) + ",";
  json += "\"foto4\":"  + String(lastBR) + ",";
  json += "\"serwo1\":" + String(currentServoAngle) + ",";
  json += "\"serwo2\":" + String(motorMovingHoriz ? 1 : 0);
  json += "}";

  int code = http.POST(json);
  Serial.print("POST kod: ");
  Serial.print(code);
  Serial.print(" | dane: ");
  Serial.println(json);

  http.end();
}


void setup() {
  Serial.begin(115200);

  solarServo.attach(SERVO_PIN);
  moveServoToAngle(currentServoAngle);

  pinMode(MOTOR_IN1, OUTPUT);
  pinMode(MOTOR_IN2, OUTPUT);
  pinMode(MOTOR_PWM, OUTPUT);
  analogWrite(MOTOR_PWM, 0);

  analogReadResolution(12);
  analogSetAttenuation(ADC_11db);
  analogSetPinAttenuation(LDR_TOP_LEFT_PIN,     ADC_11db);
  analogSetPinAttenuation(LDR_TOP_RIGHT_PIN,    ADC_11db);
  analogSetPinAttenuation(LDR_BOTTOM_LEFT_PIN,  ADC_11db);
  analogSetPinAttenuation(LDR_BOTTOM_RIGHT_PIN, ADC_11db);

  WiFi.begin(ssid, password);
  Serial.print("Laczenie z WiFi");
  unsigned long startAttempt = millis();
  while (WiFi.status() != WL_CONNECTED && millis() - startAttempt < 10000) {
    delay(500);
    Serial.print(".");
  }
  if (WiFi.status() == WL_CONNECTED) {
    Serial.println("\nPolaczono: " + WiFi.localIP().toString());
  } else {
    Serial.println("\nBrak WiFi - tracker dziala bez wysylki");
  }

  Serial.println("Init done");
}

void loop() {
  updateTracking();

  unsigned long now = millis();
  if (now - lastSendTime >= SEND_INTERVAL_MS) {
    lastSendTime = now;
    sendToServer();
  }

  delay(50); // ~20 Hz trackera
}
